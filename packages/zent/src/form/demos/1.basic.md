```jsx
import {
	FormInputField,
	Form,
	FormStrategy,
	Validators,
	Pop,
	Icon,
	Button,
	Notify,
	ValidateOption,
} from 'zent';

function Component() {
	const form = Form.useForm(FormStrategy.View);
	const [disabled, setDisabled] = React.useState(false);
	const [initializing, setInitializing] = React.useState(false);

	const initialize = React.useCallback(() => {
		setInitializing(true);
		setTimeout(() => {
			form.initialize({
				name: 'zent',
				password: 'zenttnez',
				email: 'zent@youzan.com',
			});
			setInitializing(false);
		}, 1000);
	}, [form]);

	const validate = React.useCallback(async () => {
		const results = await form.validate(
			ValidateOption.IncludeUntouched |
				ValidateOption.IncludeChildrenRecursively
		);
		if (form.isValid()) {
			Notify.success('{i18n.success}');
		}
	}, [form]);

	const onSubmit = React.useCallback(form => {
		console.log(form.getValue());
		Notify.info('{i18n.submitting}');
		return new Promise((resolve, reject) => {
			setTimeout(() => {
				Math.random() > 0.5 ? resolve() : reject();
			}, 1000);
		});
	}, []);

	const onSubmitSuccess = React.useCallback(
		() => Notify.success('{i18n.success}'),
		[]
	);

	const onSubmitFail = React.useCallback(() => Notify.error('{i18n.fail}'), []);

	return (
		<>
			<Form
				layout="horizontal"
				form={form}
				disabled={disabled}
				onSubmit={onSubmit}
				onSubmitSuccess={onSubmitSuccess}
				onSubmitFail={onSubmitFail}
			>
				<FormInputField
					name="name"
					label={
						<span>
							{i18n.name}&nbsp;
							<Pop trigger="hover" content="{i18n.nameTip}" centerArrow>
								<Icon type="error-circle-o" />
							</Pop>:
						</span>
					}
					validators={[
						Validators.minLength(5, '{i18n.nameMinLength}'),
						Validators.maxLength(25, '{i18n.nameMaxLength}'),
					]}
					helpDesc="{i18n.nameHelpdesc}"
					required="{i18n.required}"
				/>
				<FormInputField
					name="password"
					type="password"
					label="{i18n.password}:"
					helpDesc={
						<span>
							{i18n.pwHelpDesc}
							<a href="https://youzan.com" target="_blank">
								{i18n.link}
							</a>
						</span>
					}
					validateOccasion={
						Form.ValidateOccasion.Blur | Form.ValidateOccasion.Change
					}
					validators={[
						Validators.pattern(/^[a-zA-Z0-9]+$/, '{i18n.passwordPattern}'),
					]}
					notice="{i18n.notice}"
					required="{i18n.required}"
				/>
				<FormInputField
					className="demo-form-basic-email"
					label="E-Mail:"
					name="email"
					validators={[
						Validators.required('{i18n.required}'),
						Validators.email('{i18n.email}'),
					]}
				/>
				<Button htmlType="submit" loading={form.isSubmitting}>
					{i18n.submit}
				</Button>
				<Button htmlType="reset">{i18n.reset}</Button>
				<Button onClick={validate}>{i18n.validate}</Button>
				<Button onClick={form.model.clear.bind(form.model)}>
					{i18n.clear}
				</Button>
				<Button loading={initializing} onClick={initialize}>
					{i18n.initialize}
				</Button>
			</Form>
			<Button
				style={{ marginTop: 24 }}
				type={disabled ? 'default' : 'error'}
				onClick={() => setDisabled(prev => !prev)}
			>
				{disabled ? '{i18n.enable}' : '{i18n.disable}'}
			</Button>
		</>
	);
}

ReactDOM.render(<Component />, mountNode);
```
